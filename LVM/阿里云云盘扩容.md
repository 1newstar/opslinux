```
yum install -y lvm2
fdisk -l
fdisk /dev/vdb



pvcreate /dev/vdb1
vgcreate vg_group /dev/vdb1
vgdisplay
lvcreate -L 199G -n vg_data vg_group 
mkfs.xfs /dev/vg_group/vg_data
mkdir -p /data0/
echo "/dev/mapper/vg_group-vg_data /data0/ xfs defaults        0 0" >> /etc/fstab
mount -a
```
